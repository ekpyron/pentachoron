#version 420 core

layout(location = 0) in vec3 vertex;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec3 tangent;
layout(location = 3) in vec2 texcoord;

layout (binding = 0, std140) uniform CameraBlock {
       mat4x4 projmat;
       mat4x4 vmat;
       mat4x4 vmatinv;
       vec4 projinfo;
       uvec2 viewport;
       vec2 invviewport;
       vec4 center;
       vec4 eye;
       float farClipPlane;
       float nearClipPlane;
};

out vec2 uv;

uniform mat4 mvmat;
uniform mat3 normalmat;

out vec3 fTangent;
out vec3 fBitangent;
out vec3 fNormal;

void main (void)
{
	fTangent = tangent;
	fNormal = normal;
	fBitangent = cross (normal, tangent);
	uv = texcoord;
	gl_Position = projmat * mvmat * vec4 (vertex, 1.0);
}
