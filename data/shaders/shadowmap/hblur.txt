#version 420 core

layout (binding = 0) uniform sampler2D image;
layout (binding = 1) uniform samplerBuffer buffer;
uniform vec2 invviewport;

layout (location = 0) out vec2 color;

vec2 getmoments (vec2 coord)
{
	float depth;
	depth = texture2D (image, coord).x;
	return vec2 (depth, depth * depth);
}

void main(void)
{
	vec2 c;
	c = getmoments (gl_FragCoord.xy * invviewport)
	    * texelFetch (buffer, 0).r;

	for (int i = 1; i < textureSize (buffer); i++)
	{
		vec2 coord;
		vec2 weightoffset = texelFetch (buffer, i).rg;

		coord = gl_FragCoord.xy + vec2 (weightoffset.g, 0);
		c += getmoments (coord * invviewport)
		     * weightoffset.r;

		coord = gl_FragCoord.xy - vec2 (weightoffset.g, 0);
		c += getmoments (coord * invviewport)
		     * weightoffset.r;
	}
	color = c;
}
