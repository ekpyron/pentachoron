struct Parameter {
	unsigned int model;
	float param1;
	float param2;
	float reflect;
};

void ReadParameter (out struct Parameter param, in int offset)
{
	vec4 tmp;
	tmp = texelFetch (parametertex, offset);
	param.model = floatBitsToUint (tmp.x);
	param.param1 = tmp.y;
	param.param2 = tmp.z;
	param.reflect = tmp.w;
}
