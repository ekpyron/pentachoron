#version 420 core
layout (location = 0) out vec4 color;
layout (binding = 0) uniform sampler2D normalbuffer;
layout (binding = 0, std140) uniform CameraBlock {
       mat4x4 projmat;
       mat4x4 vmat;
       mat4x4 vmatinv;
       vec4 projinfo;
       uvec2 viewport;
       vec2 invviewport;
       vec4 center;
       vec4 eye;
       float farClipPlane;
       float nearClipPlane;
};

void main (void)
{
	vec2 coord;
	coord = gl_FragCoord.xy / viewport;
	color = texture2D (normalbuffer, coord) * 2 - 1;
}
