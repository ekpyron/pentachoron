#version 420 core
layout(location = 0) out vec4 color;
layout(binding = 0) uniform sampler2D tex;
uniform uvec2 viewport;

void main (void)
{
	vec2 coord;
	float edge;
	coord = gl_FragCoord.xy / viewport;
	edge = texture (tex, coord).r;
	color = edge * vec4 (1, 1, 1, 1);
}
